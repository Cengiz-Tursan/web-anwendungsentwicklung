public class A extends Object {
	protected Number doIt (Object a, String b) throws IOException { return "X"; }
}
 
public class B extends A {
	// method overriding einer virtuellen Methode ==> Polymorphismus
	public Integer doIt (Object c, String d) { return d.length(); }
 
	// Method overloading ==> kein Polymorphismus
	public Integer doIt (String c, String d) { ... }
 
	// failed attempt of method overriding
	public Object doIt (Object c, String d) { ... }
 
	// failed attempt of method overriding
	Number doIt (Object a, String b) throws IOException { ... }
}

// Aufruf mit Laufzeitargumenten: "abc" "xyz"
static public void main (String[] args) {
	final A[] things = new A[] { new A(), new B()};
 
	for (final String arg : args) {
		for (final Thing thing : things) {
			System.out.println(thing.doIt(null, arg));
		}
	}
}
